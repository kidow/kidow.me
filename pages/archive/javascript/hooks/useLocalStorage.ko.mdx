### Copy Code

```tsx filename="services/hooks/index.tsx"
const useLocalStorage = (key: 'item1' | 'item2') => {
  const item = window.localStorage.getItem(key)
  const setItem = (value: string) => window.localStorage.setItem(value)

  return [item, setItem]
}
```

### Usage

```ts
const [item, setItem] = useLocalStorage()
```
