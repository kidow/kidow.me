```tsx filename="components/SkipLink/index.tsx"
import type { FC } from 'react'
import Link from 'next/link'

export interface Props extends ReactProps {
  href: string
}

const SkipLink: FC<Props> = ({ children, href }) => {
  return (
    <Link
      href={href}
      className="absolute left-[-10000px] top-auto h-px w-px overflow-hidden focus:static focus:h-auto focus:w-auto"
    >
      {children}
    </Link>
  )
}

export default SkipLink
```
