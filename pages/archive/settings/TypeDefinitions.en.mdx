# Type Definitions

### When assigning a type to a variable added to `process.env`

```ts filename="types/global.d.ts"
declare namespace NodeJS {
  interface Process {
    env: ProcessEnv
  }
  interface ProcessEnv {
    NODE_ENV: string
  }
}
```

### When assigning variable types to `window` and `document` objects

```ts filename="types/global.d.ts"
export {}
declare global {
  // Add window object variable type
  interface Window {}
  // Add document object variable type
  interface Document {}
}
```
